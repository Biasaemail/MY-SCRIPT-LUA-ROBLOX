FeatureLogic.SiangHari = function()
   -- Sumber: SIANG.txt
   local lighting = game:GetService("Lighting")
        local runService = game:GetService("RunService")
        local TARGET_TIME = 12.0
        local TARGET_BRIGHTNESS = 0.9
        local TARGET_AMBIENT = Color3.fromRGB(190, 190, 190)
        local TARGET_OUTDOOR_AMBIENT = Color3.fromRGB(190, 190, 190)
        local FOG_DISTANCE = 9e9
        local function applyBrightSettings()
                if lighting.ClockTime ~= TARGET_TIME then lighting.ClockTime = TARGET_TIME end
                if lighting.Brightness ~= TARGET_BRIGHTNESS then lighting.Brightness = TARGET_BRIGHTNESS end
                if lighting.Ambient ~= TARGET_AMBIENT then lighting.Ambient = TARGET_AMBIENT end
                if lighting.OutdoorAmbient ~= TARGET_OUTDOOR_AMBIENT then lighting.OutdoorAmbient = TARGET_OUTDOOR_AMBIENT end
                if lighting.FogEnd ~= FOG_DISTANCE then lighting.FogEnd = FOG_DISTANCE end
                local targetFogStart = FOG_DISTANCE - 1
                if lighting.FogStart ~= targetFogStart then lighting.FogStart = targetFogStart end
                local neutralColorShift = Color3.fromRGB(0, 0, 0)
                if lighting.ColorShift_Top ~= neutralColorShift then lighting.ColorShift_Top = neutralColorShift end
        end
        local connection = runService.Heartbeat:Connect(applyBrightSettings)
        applyBrightSettings()
        print("Fitur Siang Hari diaktifkan.")
   -- Script ini akan berjalan terus, cocok untuk tipe 'Execute' karena efeknya permanen di lighting.
end

FeatureLogic.AntiKick = function()
   -- Sumber: anti kick tower of hell.txt
   pcall(function()
       local registry = getreg()
       for i, v in pairs(registry) do
           if type(v) == "function" then
               local info = debug.getinfo(v)
               if info.name == "kick" then
                   hookfunction(v, function(...) return nil end)
                   print("Anti Kick: Hooked 'kick' function.")
               end
           end
       end
   end)
end

        local function executeScriptByCharacterType()
                local character = player.Character or player.CharacterAdded:Wait()

                        local humanoid = character.Humanoid
                        local rigType = humanoid.RigType
                        if rigType == Enum.HumanoidRigType.R6 then
                                print("Karakter R6 terdeteksi. Menjalankan skrip R6...")
                                local successR6, errorR6 = pcall(function() loadstring(game:HttpGet("https://pastefy.app/wa3v2Vgm/raw"))() end)
                                if not successR6 then warn("Gagal menjalankan skrip R6:", errorR6) end
                        elseif rigType == Enum.HumanoidRigType.R15 then
                                print("Karakter R15 terdeteksi. Menjalankan skrip R15...")
                                local successR15, errorR15 = pcall(function() loadstring(game:HttpGet("https://pastefy.app/YZoglOyJ/raw"))() end)
                                if not successR15 then warn("Gagal menjalankan skrip R15:", errorR15) end
                        else
                                warn("Tipe rig karakter tidak diketahui:", rigType)
                        end
                else
                        warn("Karakter atau Humanoid tidak ditemukan untuk pemain lokal.")
                end
        end
        if player.Character then
                executeScriptByCharacterType()
        else
                player.CharacterAdded:Connect(function() executeScriptByCharacterType() end)
        end
        player.CharacterAppearanceLoaded:Connect(function()
                task.wait(0.5)
                executeScriptByCharacterType()
        end)
        print("Skrip deteksi tipe karakter (R6/R15) telah dimuat.")
end

       if UserInputService and not UserInputService:IsTenFootInterface() then
           local ok, currentQuality = pcall(function() return UserInputService:GetQualityLevel() end)
           if ok and currentQuality ~= targetQualityLevel then pcall(UserInputService.SetQualityLevel, UserInputService, targetQualityLevel); log("Attempted to set QualityLevel to " .. tostring(targetQualityLevel) .. " via UserInputService.") end
       end
   end, "set rendering quality")
   local particleDisabledCount = 0
   local function processParticle(obj)

       local processed = false
       if obj:IsA("ParticleEmitter") or obj:IsA("Smoke") or obj:IsA("Fire") or obj:IsA("Explosion") then

   try(function()

       local disabledEffectCount = 0
       for _, effect in ipairs(Lighting:GetChildren()) do if visualEffectsToManage[effect.ClassName] and effect.Enabled then effect.Enabled = false; disabledEffectCount = disabledEffectCount + 1 end end

   local function getScreenSizeApproximation(part, cam)
       if not part or not cam then return 0 end
       local distance = (cam.CFrame.Position - part.Position).Magnitude
       if distance == 0 then return 1 end
       local radius = math.max(part.Size.X, part.Size.Y, part.Size.Z) / 2
       if radius == 0 then return 0 end
       return (radius / distance) * (1 / math.tan(math.rad(cam.FieldOfView / 2)))
   end
   local function isPartRelevantForLOD(part)
       if not part:IsA("BasePart") or part.Anchored or part:IsA("Terrain") then return false end
       if CollectionService:HasTag(part, PLAYER_CHARACTER_TAG) or CollectionService:HasTag(part, IMPORTANT_NO_LOD_TAG) then return false end
       if player.Character and part:IsDescendantOf(player.Character) then return false end
       return true
   end
   local onPartRemovedFromLOD
   local function updatePartLOD(part, cameraPos, cam, deltaTime)
       if not part or not part.Parent or not part:IsDescendantOf(workspace) then return false end
       local distance = (cameraPos - part.Position).Magnitude
       local screenSize = getScreenSizeApproximation(part, cam)
       local targetTransparency = 0
       local dynamicMaxDistance = LOD_MAX_DISTANCE_BASE + (part.Size.Magnitude * 0.5)
       if distance > dynamicMaxDistance or screenSize < LOD_MIN_SIZE_ON_SCREEN then targetTransparency = 1 else targetTransparency = 0 end
       local state = partLODStates[part]
       if not state then state = { currentT = part.LocalTransparencyModifier, targetT = targetTransparency, transitionTime = 0 }; partLODStates[part] = state end
       state.targetT = targetTransparency
       if state.currentT ~= state.targetT then
           state.transitionTime = math.min(state.transitionTime + deltaTime, LOD_TRANSITION_DURATION)
           local alpha = state.transitionTime / LOD_TRANSITION_DURATION
           part.LocalTransparencyModifier = state.currentT + (state.targetT - state.currentT) * alpha
           if state.transitionTime >= LOD_TRANSITION_DURATION then part.LocalTransparencyModifier = state.targetT; state.currentT = state.targetT; state.transitionTime = 0 end
       else state.transitionTime = 0; state.currentT = part.LocalTransparencyModifier end
       return true
   end
   local function onPartAddedForLOD(part)
       if isPartRelevantForLOD(part) and not partsForLOD[part] then
           partsForLOD[part] = true; partLODStates[part] = { currentT = part.LocalTransparencyModifier, targetT = part.LocalTransparencyModifier, transitionTime = 0 }
           CollectionService:AddTag(part, LOD_MANAGED_TAG); table.insert(partListForIteration, part); lodPartCount = lodPartCount + 1
           local ancestryConn = part.AncestryChanged:Connect(function(_, newParent) if not newParent then if partsForLOD[part] then onPartRemovedFromLOD(part) end end end)
           partAncestryConnections[part] = ancestryConn; table.insert(activeConnections, ancestryConn)
           if DEBUG_MODE then log("Dynamically added part to LOD: " .. part.Name .. ". Total LOD parts: " .. lodPartCount) end
       end
   end
   onPartRemovedFromLOD = function(part)
       if partsForLOD[part] then
           partsForLOD[part] = nil; partLODStates[part] = nil
           if CollectionService:HasTag(part, LOD_MANAGED_TAG) then CollectionService:RemoveTag(part, LOD_MANAGED_TAG) end
           for i = #partListForIteration, 1, -1 do
               if partListForIteration[i] == part then
                   table.remove(partListForIteration, i)
                   if currentLODIterationIndex > i and currentLODIterationIndex > 1 then currentLODIterationIndex = currentLODIterationIndex - 1 elseif currentLODIterationIndex > #partListForIteration and #partListForIteration > 0 then currentLODIterationIndex = 1 end
                   break
               end
           end
           lodPartCount = lodPartCount - 1
           if partAncestryConnections[part] then
               local connToRemove = partAncestryConnections[part]; try(connToRemove.Disconnect, "disconnect AncestryChanged for " .. part.Name); partAncestryConnections[part] = nil
               for i = #activeConnections, 1, -1 do if activeConnections[i] == connToRemove then table.remove(activeConnections, i); break end end
           end
           if DEBUG_MODE then log("Dynamically removed part from LOD: " .. part.Name .. ". Total LOD parts: " .. lodPartCount) end
       end

                       if currentLODIterationIndex > #partListForIteration then currentLODIterationIndex = 1 end
                       if #partListForIteration == 0 then break end
                       local currentPart = partListForIteration[currentLODIterationIndex]
                       if currentPart then if not updatePartLOD(currentPart, cameraPos, camera, deltaTime) then onPartRemovedFromLOD(currentPart) else currentLODIterationIndex = currentLODIterationIndex + 1 end
                       else table.remove(partListForIteration, currentLODIterationIndex); log("Removed nil entry from LOD iteration list."); lodPartCount = math.max(0, lodPartCount - 1) end
                   end
                   lastLODUpdateTime = 0
               end)
               table.insert(activeConnections, lodHeartbeatConnection)
               log("Advanced LOD system enabled with " .. lodPartCount .. " parts.")
           else log("No parts found for LOD processing, LOD system idle.") end
       end, "initialize Advanced LOD system")
   end
   try(function()
       local terrain = workspace:FindFirstChildOfClass("Terrain")
       if terrain then
           if OPTIMIZATION_LEVEL == "High" then terrain.WaterWaveSize = 0; terrain.WaterWaveSpeed = 0; terrain.WaterReflectance = 0; terrain.WaterTransparency = 1; terrain.Decoration = false; log("Aggressive terrain optimizations applied.")
           elseif OPTIMIZATION_LEVEL == "Medium" then terrain.WaterWaveSize = terrain.WaterWaveSize * 0.3; terrain.WaterWaveSpeed = terrain.WaterWaveSpeed * 0.3; terrain.WaterReflectance = terrain.WaterReflectance * 0.2; terrain.WaterTransparency = math.min(1, terrain.WaterTransparency + 0.5); log("Medium terrain optimizations applied.") end
       else log("No Terrain found to modify.") end
   end, "optimize terrain")
   local otherEffectsCount = 0
   local function processOtherEffect(obj)
       if not obj or not obj.Parent then return false end; local processed = false

       end
       return processed
   end

   if (OPTIMIZATION_LEVEL == "High") then
       local soundOptimizedCount = 0
       local function optimizeSound(sound)
           if not sound or not sound.Parent then return false end; local optimized = false
           if sound:IsA("Sound") and not CollectionService:HasTag(sound, IMPORTANT_SOUND_TAG) then
               local MAX_HEARING_DISTANCE_NON_ESSENTIAL = 80
               if sound.RollOffMode == Enum.RollOffMode.InverseTapered or sound.MaxDistance > MAX_HEARING_DISTANCE_NON_ESSENTIAL * 1.5 then sound.MaxDistance = math.min(sound.MaxDistance, MAX_HEARING_DISTANCE_NON_ESSENTIAL); sound.MinDistance = math.min(sound.MinDistance, MAX_HEARING_DISTANCE_NON_ESSENTIAL * 0.2); soundOptimizedCount = soundOptimizedCount + 1; optimized = true end
           end
           return optimized
       end
       try(function()
           for _, d in ipairs(workspace:GetDescendants()) do if d:IsA("Sound") then optimizeSound(d) end end
           local conn = workspace.DescendantAdded:Connect(function(d) if d:IsA("Sound") then task.defer(optimizeSound, d) end end); table.insert(activeConnections, conn)
           if soundOptimizedCount > 0 then log("Optimized " .. soundOptimizedCount .. " initial sounds.") end
       end, "optimize sounds")
   end
   script.Destroying:Connect(function()
       log("Performing full cleanup of AntiLag script V3.3...")
       cleanupConnections()
       for part, conn in pairs(partAncestryConnections) do if conn and typeof(conn.Disconnect) == "function" then try(conn.Disconnect, "disconnect lingering AncestryChanged for " .. (part and part.Name or "UnknownPart")) end end
       partAncestryConnections = {}
       if lodHeartbeatConnection and lodHeartbeatConnection.Connected then lodHeartbeatConnection:Disconnect() end
       log("AntiLag script cleanup complete. Visuals may need a character reset to fully revert.")
   end)
end

-- [14] PLAYER ESP (Script lengkap, bisa di-toggle)
FeatureLogic.PlayerESP = function()
   -- Sumber: ESP HIGHLIGHT .txt
   -- Seluruh kode dari ESP HIGHLIGHT .txt akan dimasukkan di sini saat dijalankan.
   local Players = game:GetService("Players")
   local Teams = game:GetService("Teams")
   local LocalPlayer = Players.LocalPlayer
   local ENEMY_COLOR = Color3.fromRGB(255, 0, 0)
   local NEUTRAL_COLOR = Color3.fromRGB(255, 255, 0)
   local OUTLINE_COLOR = Color3.fromRGB(255, 255, 255)
   local FILL_TRANSPARENCY = 0.5
   local OUTLINE_TRANSPARENCY = 0
   local HIGHLIGHT_NAME = "PlayerESP_Highlight_v2"
   local playerConnections = {}
   local function areTeamsActive() return #Teams:GetTeams() > 0 end
   local function isHighlightTarget(player)
       if not player or player == LocalPlayer then return false end
       local character = player.Character; if not character then return false end
       local humanoid = character:FindFirstChildOfClass("Humanoid"); if not humanoid or humanoid:GetState() == Enum.HumanoidStateType.Dead then return false end
       if areTeamsActive() then
           local localPlayerTeam = LocalPlayer.Team; local targetPlayerTeam = player.Team
           if localPlayerTeam and targetPlayerTeam and localPlayerTeam ~= targetPlayerTeam then return true else return false end
       else return true end
   end
   local function updatePlayerHighlight(player)
       local character = player.Character; local existingHighlight = character and character:FindFirstChild(HIGHLIGHT_NAME)
       if isHighlightTarget(player) then
           if not character then return end
           if not existingHighlight then
               existingHighlight = Instance.new("Highlight"); existingHighlight.Name = HIGHLIGHT_NAME
               existingHighlight.Adornee = character; existingHighlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
               existingHighlight.Parent = character
           end
           local isEnemy = areTeamsActive() and player.Team ~= LocalPlayer.Team
           existingHighlight.FillColor = isEnemy and ENEMY_COLOR or NEUTRAL_COLOR
           existingHighlight.OutlineColor = OUTLINE_COLOR; existingHighlight.FillTransparency = FILL_TRANSPARENCY
           existingHighlight.OutlineTransparency = OUTLINE_TRANSPARENCY; existingHighlight.Enabled = true
       else if existingHighlight then existingHighlight:Destroy() end end
   end
   local function setupPlayerConnections(player)
       if playerConnections[player] then return end; local connections = {}; playerConnections[player] = connections
       connections.CharacterAdded = player.CharacterAdded:Connect(function(character)
           task.defer(updatePlayerHighlight, player)
           local humanoid = character:WaitForChild("Humanoid", 2)
           if humanoid then connections.Died = humanoid.Died:Connect(function() task.defer(updatePlayerHighlight, player) end) end
       end)
       connections.CharacterRemoving = player.CharacterRemoving:Connect(function(character)
           if connections.Died and connections.Died.Connected then connections.Died:Disconnect(); connections.Died = nil end
           task.defer(updatePlayerHighlight, player)
       end)
       if player:FindFirstChild("Team") or player:IsA("Player") then connections.TeamChanged = player:GetPropertyChangedSignal("Team"):Connect(function() updatePlayerHighlight(player) end) end
       if player.Character then
           task.defer(updatePlayerHighlight, player)
           local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
           if humanoid and humanoid:GetState() ~= Enum.HumanoidStateType.Dead then connections.Died = humanoid.Died:Connect(function() task.defer(updatePlayerHighlight, player) end) end
       end
   end
   local function cleanupPlayerConnections(player)
       if not playerConnections[player] then return end
       for _, connection in pairs(playerConnections[player]) do if connection and typeof(connection) == "RBXScriptConnection" and connection.Connected then connection:Disconnect() end end
       playerConnections[player] = nil
       local character = player.Character; if character then local highlight = character:FindFirstChild(HIGHLIGHT_NAME); if highlight then highlight:Destroy() end end
   end
   local function updateAllPlayersHighlights() for _, player in ipairs(Players:GetPlayers()) do if player ~= LocalPlayer then updatePlayerHighlight(player) end end end
   local localPlayerTeamChangedConn = LocalPlayer:GetPropertyChangedSignal("Team"):Connect(updateAllPlayersHighlights)
   for _, player in ipairs(Players:GetPlayers()) do if player ~= LocalPlayer then setupPlayerConnections(player) end end
   local playerAddedConn = Players.PlayerAdded:Connect(function(player) task.wait(0.5); setupPlayerConnections(player) end)
   local playerRemovingConn = Players.PlayerRemoving:Connect(function(player) cleanupPlayerConnections(player) end)
   script.Destroying:Connect(function()
       print("Highlight script cleaning up...")
       if playerAddedConn and playerAddedConn.Connected then playerAddedConn:Disconnect() end
       if playerRemovingConn and playerRemovingConn.Connected then playerRemovingConn:Disconnect() end
       if localPlayerTeamChangedConn and localPlayerTeamChangedConn.Connected then localPlayerTeamChangedConn:Disconnect() end
       for player, _ in pairs(playerConnections) do cleanupPlayerConnections(player) end
       print("Highlight script cleanup complete.")
   end)
   print("Event-Driven Player Highlighting Script Initialized (v2)")
end

local function makeDraggable(guiObject, triggerObject)
   local dragging = false
   local dragInput, lastPosition, startPosition

   triggerObject.InputBegan:Connect(function(input)
       if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
           dragging = true
           lastPosition = input.Position
           startPosition = guiObject.Position
           
           input.Changed:Connect(function()
               if input.UserInputState == Enum.UserInputState.End then
                   dragging = false
               end
           end)
       end
   end)

   triggerObject.InputChanged:Connect(function(input)
       if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and dragging then
           local delta = input.Position - lastPosition
           guiObject.Position = UDim2.new(
               startPosition.X.Scale, startPosition.X.Offset + delta.X,
               startPosition.Y.Scale, startPosition.Y.Offset + delta.Y
           )
       end
   end)
end
